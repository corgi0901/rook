program = expr ("," expr | "\n" expr)*
expr = ident "=" compare | ifExpr | whileExpr | block | compare
ifExpr = "if" "(" compare ")" ("\n")? expr (("\n")? "else" ("\n")? expr)?
whileExpr = "while" "(" compare ")" ("\n")? expr
block = "{" ("\n")? expr ("," expr | "\n" expr)* ("\n")? "}"
compare = add | add ("==" | "!=" | "<" | "<=" | ">" | ">=")  add
add = mul ("+" mul | "-" mul)*
mul = unary ("*" unary | "/" unary)*
unary = ("+" | "-")? primary
primary = ident | num | "(" compare ")"
